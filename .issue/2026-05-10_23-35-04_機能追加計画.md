        - codex_home/
            - .codex_generator/
                - skills/
                - AGENTS.md
                - auth.json
                - config.toml
            - .codex_validator/
                - skills/
                - AGENTS.md
                - auth.json
                - config.toml
        - output_json_schema/
        - data_source/
        - users_data/
            - [user_name]/
                - .codex_generator/
                - .codex_validator/
                - serrions/
                    - [session_id]/
                        - generator_workspace/
                            - artifacts/
                            - tmp/
                        - validator_workspace/
                            - tmp/
                        - saved_artifacts/

        - server:
            - users_data_dir
            - timeout_seconds
        - generator:
            - max_retries
            - codex_home_dir
            - output_schema
        - validator:
